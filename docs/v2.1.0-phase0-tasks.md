```python
import socket, sys
def run_sidecar_only():
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.bind(("127.0.0.1", 0))  # OS 分配端口
    port = sock.getsockname()[1]
    sock.close()  # 立即释放，uvicorn 重新绑定
    sys.stdout.write(f"PORT={port}\n")
    sys.stdout.flush()
    # 生成随机 token
    token = secrets.token_hex(32)
    sys.stdout.write(f"TOKEN={token}\n")
    sys.stdout.flush()
    os.environ["NEBULA_PORT"] = str(port)
    os.environ["NEBULA_TOKEN"] = token
    uvicorn.run(app, host="127.0.0.1", port=port)
```

**2.2 Tauri sidecar spawn**
```rust
// src-tauri/src/sidecar.rs
use std::process::{Command, Stdio};
use tauri::Emitter;

pub fn spawn_sidecar(app: &tauri::AppHandle) -> Result<(u16, String), String> {
    let mut child = Command::new("python")
        .arg("launch.py")
        .env("NEBULA_SHELL", "tauri")
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .spawn()
        .map_err(|e| e.to_string())?;
    
    // 读取 stdout 解析 PORT= 和 TOKEN=
    let stdout = child.stdout.take().unwrap();
    let (port, token) = parse_port_and_token(stdout)?;
    
    // 注入前端
    app.emit("sidecar-ready", serde_json::json!({
        "port": port,
        "token": token
    }))?;
    
    Ok((port, token))
}
```

**2.3 端口协商协议**
1. Tauri `setup` 钩子调用 `spawn_sidecar()`
2. Python: `bind(("127.0.0.1", 0))` → stdout `PORT=xxxxx\nTOKEN=yyyy\n`
3. Tauri: 读取 stdout，解析 PORT + TOKEN
4. Tauri: `window.__NEBULA_PORT__` + `window.__NEBULA_TOKEN__` 注入前端
5. 前端 `lib/api.ts`: 动态读取 API_BASE + Authorization header

```typescript
import { invoke } from '@tauri-apps/api/core'

const isTauri = '__TAURI_INTERNALS__' in window
const API_BASE = isTauri
  ? `http://127.0.0.1:${(window as any).__NEBULA_PORT__}`
  : 'http://127.0.0.1:7860'
const TOKEN = isTauri ? (window as any).__NEBULA_TOKEN__ : null

async function request<T>(method: string, path: string, body?: any): Promise<T> {
  if (isTauri) {
    // Tauri: invoke http_proxy command（CRUD 走代理）
    const result = await invoke('http_proxy', { method, path, body })
    if (!result.ok) throw new Error(result.error)
    return result.data as T
  } else {
    // 浏览器: 保留原 fetch（脱离 Tauri 独立开发）
    const headers: Record<string, string> = { 'Content-Type': 'application/json' }
    if (TOKEN) headers['Authorization'] = `Bearer ${TOKEN}`
    const res = await fetch(`${API_BASE}${path}`, {
      method, headers, body: body ? JSON.stringify(body) : undefined
    })
    const json = await res.json()
    if (!json.ok) throw new Error(json.error)
    return json.data as T
  }
}

export const apiGet = <T>(path: string) => request<T>('GET', path)
export const apiPost = <T>(path: string, body: any) => request<T>('POST', path, body)
export const apiPut = <T>(path: string, body: any) => request<T>('PUT', path, body)
export const apiDelete = <T>(path: string) => request<T>('DELETE', path)
```

**3.2 Rust http_proxy command**
```rust
// src-tauri/src/ipc.rs
use reqwest::Client;
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize)]
pub struct ProxyResponse {
    pub ok: bool,
    pub data: Option<serde_json::Value>,
    pub error: Option<String>,
}

#[tauri::command]
pub async fn http_proxy(
    method: String,
    path: String,
    body: Option<serde_json::Value>,
    state: tauri::State<'_, AppState>,
) -> Result<ProxyResponse, String> {
    let url = format!("http://127.0.0.1:{}{}", state.port, path);
    let client = Client::new();
    let mut req = match method.as_str() {
        "GET" => client.get(&url),
        "POST" => client.post(&url).json(&body),
        "PUT" => client.put(&url).json(&body),
        "DELETE" => client.delete(&url),
        _ => return Err("Unsupported method".into()),
    };
    req = req.header("Authorization", format!("Bearer {}", state.token));
    
    let res = req.send().await.map_err(|e| e.to_string())?;
    let json: serde_json::Value = res.json().await.map_err(|e| e.to_string())?;
    
    Ok(ProxyResponse {
        ok: json.get("ok").and_then(|v| v.as_bool()).unwrap_or(false),
        data: json.get("data").cloned(),
        error: json.get("error").and_then(|v| v.as_str()).map(String::from),
    })
}
```

```rust
// src-tauri/src/integrity.rs
use sha2::{Sha256, Digest};
use std::fs::File;

pub fn verify_sidecar(expected_hash: &str) -> Result<(), String> {
    let sidecar_path = std::env::current_exe()
        .map_err(|e| e.to_string())?
        .parent()
        .unwrap()
        .join("resources/pangu-sidecar/pangu-nebula.exe");
    
    let mut file = File::open(&sidecar_path).map_err(|e| e.to_string())?;
    let mut hasher = Sha256::new();
    std::io::copy(&mut file, &mut hasher).map_err(|e| e.to_string())?;
    let actual_hash = format!("{:x}", hasher.finalize());
    
    if actual_hash != expected_hash {
        return Err(format!("Sidecar integrity check failed: expected {}, got {}", expected_hash, actual_hash));
    }
    Ok(())
}
```

```rust
pub fn watch_sidecar(app: &tauri::AppHandle, mut child: CommandChild) {
```

**5.3 指数退避重启**
```rust
async fn restart_with_backoff(app: &tauri::AppHandle) {
    let state = app.state::<AppState>();
    let mut retry_count = state.retry_count.lock().await;
    
    if *retry_count >= 3 {
        // 触发降级 UI
        app.emit("sidecar-degraded", ()).unwrap();
        return;
    }
    
    let delay = 2u64.pow(*retry_count as u32); // 1s → 2s → 4s
    tokio::time::sleep(Duration::from_secs(delay)).await;
    
    match spawn_sidecar(app) {
        Ok(_) => *retry_count = 0,
        Err(_) => *retry_count += 1,
    }
}
```

**5.4 优雅关闭**
```rust
pub async fn graceful_shutdown(app: &tauri::AppHandle) {
    let state = app.state::<AppState>();
    // 1. 调用 /shutdown 端点
    let _ = reqwest::Client::new()
        .post(format!("http://127.0.0.1:{}/shutdown", state.port))
        .header("Authorization", format!("Bearer {}", state.token))
        .send()
        .await;
    // 2. 等待 5s
    tokio::time::sleep(Duration::from_secs(5)).await;
    // 3. kill
    let _ = state.child.lock().await.kill();
}
```

## 回退方案

1. **Feature flag**: `NEBULA_SHELL=tauri|pywebview` 环境变量控制
2. **launch.py 保留**: 不删除原 PyWebView 入口
3. **lib/api.ts 双实现**: 保留 fetch 后备，支持脱离 Tauri 开发
4. **release.yml 保留**: PyInstaller 路线作为 fallback，Tauri 稳定 3 次发布后废弃
5. **数据零迁移**: SQLite 格式不变，回退无数据成本

**触发回退的条件**：
- Go/No-Go #3（Sidecar PoC）失败
- Go/No-Go #6（回归测试）连续 2 次失败
- 累计延期超过 4 周
